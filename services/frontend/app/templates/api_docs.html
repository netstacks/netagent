Quick Examples
# List all agents
curl -X GET "{{ request.url.scheme }}://{{ request.url.netloc }}/api/agents" \
  -H "Content-Type: application/json"

# Create a chat session
curl -X POST "{{ request.url.scheme }}://{{ request.url.netloc }}/api/chat/sessions" \
  -H "Content-Type: application/json" \
  -d '{"agent_id": 1}'

# Send a message
curl -X POST "{{ request.url.scheme }}://{{ request.url.netloc }}/api/chat/sessions/{session_id}/messages" \
  -H "Content-Type: application/json" \
  -d '{"content": "Hello, agent!"}'
import requests

BASE_URL = "{{ request.url.scheme }}://{{ request.url.netloc }}/api"

# List all agents
response = requests.get(f"{BASE_URL}/agents")
agents = response.json()

# Create a chat session
session = requests.post(f"{BASE_URL}/chat/sessions", json={
    "agent_id": 1
}).json()

# Send a message
message = requests.post(
    f"{BASE_URL}/chat/sessions/{session['id']}/messages",

const BASE_URL = '{{ request.url.scheme }}://{{ request.url.netloc }}/api';

// List all agents
const agents = await fetch(`${BASE_URL}/agents`)
  .then(r => r.json());

// Create a chat session
const session = await fetch(`${BASE_URL}/chat/sessions`, {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ agent_id: 1 })
}).then(r => r.json());

// Send a message (streaming)
const response = await fetch(`${BASE_URL}/chat/sessions/${session.id}/messages/stream`, {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ content: 'Hello, agent!' })
});
